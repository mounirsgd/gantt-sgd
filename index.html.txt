<script>
function bandeau() {
    const jours = ["DIMANCHE","LUNDI","MARDI","MERCREDI","JEUDI","VENDREDI","SAMEDI"];
    const d = new Date();
    const jour = jours[d.getDay()];
    const date = d.toLocaleDateString("fr-FR");
    const semaine = Math.ceil((d.getDate() - d.getDay() + 10) / 7);

    document.getElementById("header").textContent =
        "PRÉPARATION CHANGEMENT DE FABRICATION BOUT CHAUD — Semaine " + semaine + " — " + jour + " " + date;
}

bandeau();

function valider() {
    const ids = [
        "lig","heure","article","code","vstd","vfab","pds","cuv","vc","cp",
        "cuil","int","def","be","serie","fabfin","finsa","i1","i2","i3",
        "tps","trait","obs","cds","prod"
    ];

    let values = {};
    ids.forEach(id => {
        let v = document.getElementById(id).value.trim();
        values[id] = v === "" ? "<span class='none-value'>None</span>" : v;
    });

    let html = `
        <tr>
            <th>Lig</th><th>Heure</th><th>ARTICLE</th><th>CODE</th>
            <th colspan="2">VITESSE</th>
            <th>PDS</th>
            <th colspan="3">FEEDER</th>
            <th colspan="3">CONDUITS</th>
            <th>BE</th>
            <th colspan="3">MOULERIE</th>
            <th colspan="3">INDICES</th>
            <th>Tps</th><th>Trait</th><th>OBS</th><th>CdS</th><th>Prod</th>
        </tr>
        <tr>
            <td>${values.lig}</td>
            <td>${values.heure}</td>
            <td>${values.article}</td>
            <td>${values.code}</td>

            <td>${values.vstd}</td>
            <td>${values.vfab}</td>

            <td>${values.pds}</td>

            <td>${values.cuv}</td>
            <td>${values.vc}</td>
            <td>${values.cp}</td>

            <td>${values.cuil}</td>
            <td>${values.int}</td>
            <td>${values.def}</td>

            <td>${values.be}</td>

            <td>${values.serie}</td>
            <td>${values.fabfin}</td>
            <td>${values.finsa}</td>

            <td>${values.i1}</td>
            <td>${values.i2}</td>
            <td>${values.i3}</td>

            <td>${values.tps}</td>
            <td>${values.trait}</td>
            <td>${values.obs}</td>
            <td>${values.cds}</td>
            <td>${values.prod}</td>
        </tr>
    `;

    document.getElementById("final-table").innerHTML = html;

    document.getElementById("form-section").classList.add("hidden");
    document.getElementById("table-section").classList.remove("hidden");
}

function retour() {
    document.getElementById("table-section").classList.add("hidden");
    document.getElementById("form-section").classList.remove("hidden");
}
